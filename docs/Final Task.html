def getSpringDate(fileName,gddVal):
    df = pd.read_csv('./Data/50620-2017_GDD.csv',index_col=0)
    #print(df['Daily GDD'].values)
    gdd = df['Daily GDD']
    #print(gdd)
    c = gdd.cumsum()
    val = np.flatnonzero(np.isclose(c.values, gddVal,atol=20))[0]
    return df.iloc[val]
